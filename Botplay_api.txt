Lưu ý match_id lấy sau khi chạy lệnh curl đầu tiên

# Khởi tạo chế độ chơi với bot
curl -X POST http://localhost:5002/api/mode/bot \
  -H "Content-Type: application/json" \
  -d '{"user_id": 1}'

# Đi nước với bot
curl -X POST http://localhost:5002/api/game/bot/move \
  -H "Content-Type: application/json" \
  -d '{"match_id": 1, "move": "e2e4"}'
